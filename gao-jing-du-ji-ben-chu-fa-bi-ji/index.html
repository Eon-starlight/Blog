#include<bits/stdc++.h>//万能库
using namespace std;
int a[1000001],b[1000001],ans[1000001],tmp[1000001],al,bl,asl,tl;
string ax,bx;//开变量
int main()
{
    cin>>ax>>bx;//读入不解释
	for(int i=1;i<=ax.size();i++)a[i]=ax[ax.size()-i]-'0';
	for(int i=1;i<=bx.size();i++)b[i]=bx[bx.size()-i]-'0';
	al=ax.size();
	bl=bx.size();
    asl=al-bl+1;
    for(int i=asl;i>0;i--)//重要的部分
	{
        memset(tmp,0,sizeof(tmp));//清0
        for(int j=1;j<=bl;j++)tmp[j+i-1]=b[j];//位移
    	tl=bl+i-1;
    	int pd=0;//判断谁大
    	if(al>tl)pd=1;
    	if(al<tl)pd=-1;
    	for(int j=al;j>0;j--)
		{
			if(a[j]>tmp[j])pd=1;
			if(a[j]<tmp[j])pd=-1;
		}
        while(pd>=0)//相减计数
		{
			ans[i]++;
			if(pd==0)al=0;
    		else
    		{
        		for(int j=1;j<=al;j++)
        		{
        		    if(a[j]<b[j])
					{
						a[i+1]--;
						a[i]+=10;
					}
    		        if(a[j]>=b[j])a[j]-=b[j];
    		    }
    		    while((a[al]==0)&&(al>0))al--;
    		}
			pd=0;
    		if(al>tl)pd=1;
    		if(al<tl)pd=-1;
    		for(int j=al;j>0;j--)
			{
				if(a[j]>tmp[j])pd=1;
				if(a[j]<tmp[j])pd=-1;
			}
		}
    }
    while((ans[asl]==0)&&(asl>0))asl--;//去掉高位的0
    if(asl==0)printf("0");输出不解释
	else
	{
    	for(int i=asl;i>0;i--)printf("%d",ans[i]);
    }
	printf(" ");
	if(al==0)printf("0");
	else
	{
    	for(int i=al;i>0;i--)printf("%d",a[i]);
    }
    return 0;
}
